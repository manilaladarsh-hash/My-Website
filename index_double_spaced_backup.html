# --- Generative Wave Synthesis Suite ---

import numpy as np

from scipy.io import wavfile

from scipy import signal



SAMPLE_RATE = 44100

BPM = 95

BEAT_DUR = 60.0 / BPM

BAR_DUR = BEAT_DUR * 4



# Mix detuned sawtooths for thick pad

def synth_pad(note, dur):

    t = np.linspace(0, dur, int(dur*SAMPLE_RATE), False)

    f = 440.0 * (2.0 ** ((note - 69) / 12.0))

    wave = 0.5 * signal.sawtooth(2 * np.pi * f * t)

    wave += 0.3 * signal.sawtooth(2 * np.pi * f * 1.005 * t)

    wave += 0.3 * signal.sawtooth(2 * np.pi * f * 0.995 * t)

    # Lowpass smooth filter

    b, a = signal.butter(2, 800 / (SAMPLE_RATE / 2), 'low')

    return signal.lfilter(b, a, wave) * 0.3



# Pure square growling bassline

def bass(note, dur):

    t = np.linspace(0, dur, int(dur*SAMPLE_RATE), False)

    f = 440.0 * (2.0 ** ((note - 69) / 12.0))

    wave = signal.square(2 * np.pi * f * t)

    b, a = signal.butter(2, 300 / (SAMPLE_RATE / 2), 'low')

    return signal.lfilter(b, a, wave) * 0.5
